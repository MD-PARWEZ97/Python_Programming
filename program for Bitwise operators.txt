#Q.3Write a program for Bitwise operators 

a = 5  
b = 3  
# AND
print("Bitwise AND (a & b):", a & b)  
# OR
print("Bitwise OR (a | b):", a | b)  
# XOR
print("Bitwise XOR (a ^ b):", a ^ b)  
# NOT
print("Bitwise NOT (~a):", ~a)        
# left Shift
print("Left Shift (a << 1):", a << 1)
# Right Shift
print("Right Shift (a >> 1):", a >> 1)
