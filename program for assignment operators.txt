#Q.2 Write a program for assignment operators

# Initial value
a = 10
print("Initial value of a:", a)
# Add and assign
a += 5                                             # a = a + 5
print("After 'a += 5':", a)
# Subtract and assign
a -= 3                                            # a = a - 3
print("After 'a -= 3':", a)
# Multiply and assign
a *= 2                                         # a = a * 2
print("After 'a *= 2':", a)
# Divide and assign
a /= 4                                        # a = a / 4
print("After 'a /= 4':", a)
# Modulus and assign
a %= 3                                      # a = a % 3
print("After 'a %= 3':", a)
# Exponentiation and assign
a **= 2                                    # a = a ** 2
print("After 'a **= 2':", a)
# Floor division and assigna //= 2       # a = a // 2
print("After 'a //= 2':", a)
